async function getAddressFromCoords(lat, lng) {
  try {
    const url = `https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lng}&zoom=18&addressdetails=1`;
    const response = await fetch(url, { headers: { 'Accept-Language': 'id-ID' } });
    const data = await response.json();
    if (data && data.address) {
      const addr = data.address;
      const road = addr.road || addr.pedestrian || "";
      const village = addr.village || addr.suburb || addr.neighbourhood || "";
      const district = addr.city_district || addr.district || addr.municipality || "";
      const parts = [road, village, district].filter(p => p !== "");
      return parts.length > 0 ? parts.join(", ") : data.display_name.split(',').slice(0, 3).join(',');
    }
    return "Lokasi tidak terdeteksi";
  } catch (error) { return "Gagal memuat alamat"; }
}

function listenFirebase() {
    onValue(ref(db, "panelPJU"), (snapshot) => {
        document.getElementById("last-sync-time").innerText = "Sync: " + new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' });
        if(!snapshot.exists()) { allPanelData = []; renderList([]); updateStats(0, 0); return; }
        allPanelData = Object.entries(snapshot.val()).map(([id, val]) => ({ id, val }))
            .sort((a,b) => parseInt(a.id.split(' ')[1]) - parseInt(b.id.split(' ')[1]));
        
        let total = 0, on = 0;
        const bounds = new google.maps.LatLngBounds();
        let validCoords = 0;

        allPanelData.forEach(({id, val}) => {
            const isPanelOn = val.status_panel === "NYALA";
            total++; if(isPanelOn) on++;
            const pos = { lat: parseFloat(val.latitude), lng: parseFloat(val.longitude) };
            if(!isNaN(pos.lat) && !isNaN(pos.lng)) {
                validCoords++;
                bounds.extend(pos);
                const color = isPanelOn ? "#fbbf24" : "#94a3b8";
                const icon = { 
                  path: "M12 2C8.13 2 5 5.13 5 9c0 2.38 1.19 4.47 3 5.74V17c0 .55.45 1 1 1h6c.55 0 1-.45 1-1v-2.26c1.81-1.27 3-3.36 3-5.74 0-3.87-3.13-7-7-7zM9 21c0 .55.45 1 1 1h4c.55 0 1-.45 1-1v-1H9v1z", 
                  fillColor: color, 
                  fillOpacity: 1, 
                  strokeWeight: 2, 
                  strokeColor: "#ffffff", 
                  scale: 1.3,
                  anchor: new google.maps.Point(12, 12)
                };
                if (markers[id]) { 
                  markers[id].setPosition(pos); 
                  markers[id].setIcon(icon); 
                } else { 
                  markers[id] = new google.maps.Marker({ 
                    position: pos, 
                    map: map, 
                    icon: icon, 
                    label: { 
                      text: id, 
                      fontSize: "10px", 
                      fontWeight: "900", 
                      color: "#000", 
                      labelOrigin: new google.maps.Point(12, -15) 
                    } 
                  }); 
                }
            }
        });
        if(validCoords > 0 && !initialBoundsSet) {
          if (validCoords === 1) { map.setCenter(bounds.getCenter()); map.setZoom(16); }
          else { map.fitBounds(bounds); }
          initialBoundsSet = true;
        }
        filterPanels(); updateStats(total, on);
    });
}

    let nextNum = 1; if(snap.exists()) nextNum = Object.keys(snap.val()).length + 1;
    const customID = `PJU${nextNum}`;
    
    await set(ref(db, "panelPJU/" + customID), {
      link_api: weatherApiUrl,
      id_panel: customID, nama_panel: nama, latitude: lat, longitude: lon, alamat_detail: fullAddress,
      cuaca_id: data.weather?.[0]?.id || 800, cuaca: data.weather?.[0]?.main || 'Clear', suhu: Math.round(data.main?.temp || 0),
      jarak_pandang: data.visibility || 10000, awan: data.clouds?.all || 0, status_panel: "MATI", mode: "OTOMATIS",
      update_terakhir: new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' })
    });
    toggleModal(false); initialBoundsSet = false; showToast(`${customID} terdaftar.`);
  } catch (e) { showToast("Gagal registrasi"); } finally { document.getElementById("btnSimpan").innerText = "Simpan Perangkat"; }
}
